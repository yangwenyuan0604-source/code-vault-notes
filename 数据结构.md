# 数据结构

> 数据结构 = 数据 + 结构，是计算机存储、组织数据的方式。**程序 = 数据结构 + 算法**。

---

## 一、复杂度分析

| 复杂度 | 含义 | 示例 |
|--------|------|------|
| $O(1)$ | 常数时间 | 数组下标访问 |
| $O(\log n)$ | 对数时间 | 二分查找 |
| $O(n)$ | 线性时间 | 遍历数组 |
| $O(n \log n)$ | 线性对数 | 归并排序 |
| $O(n^2)$ | 平方时间 | 冒泡排序 |
| $O(2^n)$ | 指数时间 | 递归求斐波那契 |

> [!tip] 空间换时间
> 哈希表用空间换 $O(1)$ 查找；跳跃表用多层索引加速链表查询。

## 二、数组 (Array)

**连续内存，随机访问 $O(1)$，插入/删除 $O(n)$。**

### C 语言

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// ========== 动态数组 ==========
typedef struct {
    int *data;
    int capacity;
    int size;
} ArrayList;

ArrayList *al_create(int capacity) {
    ArrayList *al = malloc(sizeof(ArrayList));
    al->data = malloc(sizeof(int) * capacity);
    al->capacity = capacity;
    al->size = 0;
    return al;
}

void al_add(ArrayList *al, int index, int val) {
    if (index < 0 || index > al->size) return;
    if (al->size == al->capacity) {
        al->capacity *= 2;
        al->data = realloc(al->data, sizeof(int) * al->capacity);
    }
    // 后移元素
    for (int i = al->size; i > index; i--)
        al->data[i] = al->data[i - 1];
    al->data[index] = val;
    al->size++;
}

int al_remove(ArrayList *al, int index) {
    if (index < 0 || index >= al->size) return -1;
    int removed = al->data[index];
    for (int i = index; i < al->size - 1; i++)
        al->data[i] = al->data[i + 1];
    al->size--;
    return removed;
}

void al_free(ArrayList *al) {
    free(al->data);
    free(al);
}
```

### Java | 手动实现动态数组

```java
class MyArrayList {
    private int[] data;
    private int size;
    private int capacity;

    public MyArrayList(int capacity) {
        this.capacity = capacity;
        this.data = new int[capacity];
        this.size = 0;
    }

    // 尾部追加 O(1) 均摊
    public void add(int val) {
        if (size == capacity) resize(capacity * 2);
        data[size++] = val;
    }

    // 指定位置插入 O(n)
    public void add(int index, int val) {
        if (index < 0 || index > size) return;
        if (size == capacity) resize(capacity * 2);
        for (int i = size; i > index; i--)
            data[i] = data[i - 1];
        data[index] = val;
        size++;
    }

    // 删除 O(n)
    public int remove(int index) {
        if (index < 0 || index >= size) return -1;
        int removed = data[index];
        for (int i = index; i < size - 1; i++)
            data[i] = data[i + 1];
        size--;
        return removed;
    }

    // 随机访问 O(1)
    public int get(int index) {
        if (index < 0 || index >= size) return -1;
        return data[index];
    }

    public void set(int index, int val) {
        if (index >= 0 && index < size) data[index] = val;
    }

    public int size() { return size; }
    public boolean isEmpty() { return size == 0; }

    private void resize(int newCap) {
        int[] newData = new int[newCap];
        for (int i = 0; i < size; i++) newData[i] = data[i];
        data = newData;
        capacity = newCap;
    }

    // 测试入口
    public static void main(String[] args) {
        MyArrayList list = new MyArrayList(4);
        list.add(10);
        list.add(20);
        list.add(1, 15);   // [10, 15, 20]
        list.remove(0);    // [15, 20]
        System.out.println(list.get(0)); // 15
    }
}
```

> [!warning] 数组 vs ArrayList
> - 数组：固定长度，存基本类型，无额外开销
> - ArrayList：动态扩容(1.5倍)，存对象，有装箱开销

## 三、链表 (Linked List)

**非连续内存，插入/删除 $O(1)$，查找 $O(n)$。**

### 单向链表 | C

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int val;
    struct Node *next;
} Node;

Node *node_create(int val) {
    Node *n = malloc(sizeof(Node));
    n->val = val;
    n->next = NULL;
    return n;
}

// 头插法
void list_add_head(Node **head, int val) {
    Node *n = node_create(val);
    n->next = *head;
    *head = n;
}

// 删除指定值
Node *list_remove(Node *head, int val) {
    Node dummy = {0, head};
    Node *prev = &dummy, *cur = head;
    while (cur) {
        if (cur->val == val) {
            prev->next = cur->next;
            free(cur);
            cur = prev->next;
        } else {
            prev = cur;
            cur = cur->next;
        }
    }
    return dummy.next;
}

// 反转链表 (三指针)
Node *list_reverse(Node *head) {
    Node *prev = NULL, *cur = head, *next;
    while (cur) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    return prev;
}

// 快慢指针找中点
Node *list_find_mid(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;
}

// 检测环
int list_has_cycle(Node *head) {
    Node *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) return 1;
    }
    return 0;
}

void list_free(Node *head) {
    while (head) {
        Node *tmp = head;
        head = head->next;
        free(tmp);
    }
}
```

### 双向链表 | C

```c
typedef struct DNode {
    int val;
    struct DNode *prev, *next;
} DNode;

DNode *dnode_create(int val) {
    DNode *n = malloc(sizeof(DNode));
    n->val = val;
    n->prev = n->next = NULL;
    return n;
}

// 尾插入
void dlist_add_tail(DNode **head, DNode **tail, int val) {
    DNode *n = dnode_create(val);
    if (*tail) {
        (*tail)->next = n;
        n->prev = *tail;
        *tail = n;
    } else {
        *head = *tail = n;
    }
}

// 删除节点 (O(1)，已知节点指针)
void dlist_remove_node(DNode **head, DNode **tail, DNode *node) {
    if (!node) return;
    if (node->prev) node->prev->next = node->next;
    else *head = node->next;
    if (node->next) node->next->prev = node->prev;
    else *tail = node->prev;
    free(node);
}
```

### Java | 单向链表 (手动实现)

```java
class ListNode {
    int val;
    ListNode next;
    ListNode(int x) { val = x; }
}

class MyLinkedList {
    private ListNode head;

    // 头插 O(1)
    public void addFirst(int val) {
        ListNode n = new ListNode(val);
        n.next = head;
        head = n;
    }

    // 尾插 O(n)
    public void addLast(int val) {
        ListNode n = new ListNode(val);
        if (head == null) { head = n; return; }
        ListNode cur = head;
        while (cur.next != null) cur = cur.next;
        cur.next = n;
    }

    // 删除指定值 O(n)
    public void remove(int val) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy, cur = head;
        while (cur != null) {
            if (cur.val == val) {
                prev.next = cur.next;
                cur = prev.next;
            } else {
                prev = cur;
                cur = cur.next;
            }
        }
        head = dummy.next; // 恢复头
    }

    // 反转链表 (三指针)
    public void reverse() {
        ListNode prev = null, cur = head, next;
        while (cur != null) {
            next = cur.next;
            cur.next = prev;
            prev = cur;
            cur = next;
        }
        head = prev;
    }

    // 快慢指针找中点
    public ListNode findMid() {
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        return slow;
    }

    // 检测环
    public boolean hasCycle() {
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) return true;
        }
        return false;
    }

    // 打印链表
    public void print() {
        for (ListNode cur = head; cur != null; cur = cur.next)
            System.out.print(cur.val + " -> ");
        System.out.println("null");
    }

    // 测试
    public static void main(String[] args) {
        MyLinkedList list = new MyLinkedList();
        list.addFirst(3);  // 3
        list.addFirst(2);  // 2 -> 3
        list.addFirst(1);  // 1 -> 2 -> 3
        list.addLast(4);   // 1 -> 2 -> 3 -> 4
        list.remove(2);    // 1 -> 3 -> 4
        list.reverse();    // 4 -> 3 -> 1
        list.print();
    }
}
```

```java
class DoubleListNode {
    int val;
    DoubleListNode prev, next;
    DoubleListNode(int x) { val = x; }
}

class MyDoubleList {
    private DoubleListNode head, tail;

    // 尾插 O(1)
    public void addLast(int val) {
        DoubleListNode n = new DoubleListNode(val);
        if (tail != null) {
            tail.next = n;
            n.prev = tail;
            tail = n;
        } else {
            head = tail = n;
        }
    }

    // 头插 O(1)
    public void addFirst(int val) {
        DoubleListNode n = new DoubleListNode(val);
        if (head != null) {
            n.next = head;
            head.prev = n;
            head = n;
        } else {
            head = tail = n;
        }
    }

    // 删除指定节点 O(1)
    public void remove(DoubleListNode node) {
        if (node == null) return;
        if (node.prev != null) node.prev.next = node.next;
        else head = node.next;
        if (node.next != null) node.next.prev = node.prev;
        else tail = node.prev;
    }

    // 查找 O(n)
    public DoubleListNode find(int val) {
        DoubleListNode cur = head;
        while (cur != null) {
            if (cur.val == val) return cur;
            cur = cur.next;
        }
        return null;
    }

    // 正向打印
    public void printForward() {
        for (DoubleListNode cur = head; cur != null; cur = cur.next)
            System.out.print(cur.val + " <-> ");
        System.out.println("null");
    }
}
```

## 四、栈 (Stack)

**LIFO (后进先出)。push/pop/peek 均为 $O(1)$。**

### C 语言 | 数组实现

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int *data;
    int top;      // 栈顶指针
    int capacity;
} Stack;

Stack *stack_create(int capacity) {
    Stack *s = malloc(sizeof(Stack));
    s->data = malloc(sizeof(int) * capacity);
    s->top = -1;
    s->capacity = capacity;
    return s;
}

int stack_push(Stack *s, int val) {
    if (s->top == s->capacity - 1) {
        s->capacity *= 2;
        s->data = realloc(s->data, sizeof(int) * s->capacity);
    }
    s->data[++(s->top)] = val;
    return 0;
}

int stack_pop(Stack *s) {
    if (s->top == -1) return -1; // 栈空
    return s->data[(s->top)--];
}

int stack_peek(Stack *s) {
    return s->top == -1 ? -1 : s->data[s->top];
}

int stack_is_empty(Stack *s) { return s->top == -1; }

void stack_free(Stack *s) {
    free(s->data);
    free(s);
}
```

### C 语言 | 链表实现

```c
typedef struct SNode {
    int val;
    struct SNode *next;
} SNode;

SNode *snode_create(int val) {
    SNode *n = malloc(sizeof(SNode));
    n->val = val;
    n->next = NULL;
    return n;
}

void llstack_push(SNode **top, int val) {
    SNode *n = snode_create(val);
    n->next = *top;
    *top = n;
}

int llstack_pop(SNode **top) {
    if (!*top) return -1;
    SNode *tmp = *top;
    int val = tmp->val;
    *top = (*top)->next;
    free(tmp);
    return val;
}
```

```java
class MyArrayStack {
    private int[] data;
    private int top;       // 栈顶指针

    public MyArrayStack(int capacity) {
        data = new int[capacity];
        top = -1;
    }

    public boolean push(int val) {
        if (top == data.length - 1) {
            // 扩容
            int[] newData = new int[data.length * 2];
            for (int i = 0; i <= top; i++) newData[i] = data[i];
            data = newData;
        }
        data[++top] = val;
        return true;
    }

    public int pop() {
        if (top == -1) return -1;
        return data[top--];
    }

    public int peek() {
        return top == -1 ? -1 : data[top];
    }

    public boolean isEmpty() { return top == -1; }
    public int size() { return top + 1; }
}
```

## 五、队列 (Queue)

**FIFO (先进先出)。enqueue/dequeue $O(1)$。**

### C 语言 | 循环队列

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct {
    int *data;
    int head, tail;
    int capacity;
    int size;
} MyQueue;

MyQueue *queue_create(int capacity) {
    MyQueue *q = malloc(sizeof(MyQueue));
    q->data = malloc(sizeof(int) * capacity);
    q->head = 0;
    q->tail = 0;
    q->size = 0;
    q->capacity = capacity;
    return q;
}

bool queue_enqueue(MyQueue *q, int val) {
    if (q->size == q->capacity) {
        // 扩容
        int new_cap = q->capacity * 2;
        int *new_data = malloc(sizeof(int) * new_cap);
        for (int i = 0; i < q->size; i++)
            new_data[i] = q->data[(q->head + i) % q->capacity];
        free(q->data);
        q->data = new_data;
        q->head = 0;
        q->tail = q->size;
        q->capacity = new_cap;
    }
    q->data[q->tail] = val;
    q->tail = (q->tail + 1) % q->capacity;
    q->size++;
    return true;
}

int queue_dequeue(MyQueue *q) {
    if (q->size == 0) return -1;
    int val = q->data[q->head];
    q->head = (q->head + 1) % q->capacity;
    q->size--;
    return val;
}

bool queue_is_empty(MyQueue *q) { return q->size == 0; }

int queue_peek(MyQueue *q) {
    return q->size == 0 ? -1 : q->data[q->head];
}

void queue_free(MyQueue *q) {
    free(q->data);
    free(q);
}
```

### Java | 链表实现

```java
class QueueNode {
    int val;
    QueueNode next;
    QueueNode(int x) { val = x; }
}

class MyQueue {
    private QueueNode head, tail;
    private int size;

    // 入队 O(1)
    public void enqueue(int val) {
        QueueNode n = new QueueNode(val);
        if (tail != null) {
            tail.next = n;
            tail = n;
        } else {
            head = tail = n;
        }
        size++;
    }

    // 出队 O(1)
    public int dequeue() {
        if (head == null) return -1;
        int val = head.val;
        head = head.next;
        if (head == null) tail = null;
        size--;
        return val;
    }

    public int peek() { return head == null ? -1 : head.val; }
    public boolean isEmpty() { return head == null; }
    public int size() { return size; }
}
```

### Java | 循环队列 (数组实现)

```java
class MyCircularQueue {
    private int[] data;
    private int head, tail, size, capacity;

    public MyCircularQueue(int cap) {
        capacity = cap;
        data = new int[cap];
        head = tail = size = 0;
    }

    public boolean enqueue(int val) {
        if (size == capacity) return false;
        data[tail] = val;
        tail = (tail + 1) % capacity;
        size++;
        return true;
    }

    public int dequeue() {
        if (size == 0) return -1;
        int val = data[head];
        head = (head + 1) % capacity;
        size--;
        return val;
    }

    public int peek() {
        return size == 0 ? -1 : data[head];
    }

    public boolean isEmpty() { return size == 0; }
    public boolean isFull()  { return size == capacity; }
}
```

## 六、哈希表 (Hash Table)

**键值映射，查找/插入/删除均摊 $O(1)$。核心：哈希函数 + 冲突解决。**

### 冲突解决策略

| 方法 | 原理 | 特点 |
|------|------|------|
| 链地址法 | 每个桶存链表 | Java HashMap 默认 |
| 开放地址法 | 线性/二次探测 | Python dict 早期 |

### C 语言 | 链地址法

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define CAPACITY 16
#define LOAD_FACTOR 0.75

typedef struct Entry {
    char *key;
    int value;
    struct Entry *next;
} Entry;

typedef struct {
    Entry **buckets;
    int capacity;
    int size;
} HashMap;

unsigned int hash(const char *key, int capacity) {
    unsigned int h = 0;
    while (*key) h = h * 31 + *key++;
    return h % capacity;
}

HashMap *hm_create() {
    HashMap *hm = malloc(sizeof(HashMap));
    hm->capacity = CAPACITY;
    hm->size = 0;
    hm->buckets = calloc(CAPACITY, sizeof(Entry*));
    return hm;
}

Entry *hm_find_entry(Entry *head, const char *key) {
    while (head) {
        if (strcmp(head->key, key) == 0) return head;
        head = head->next;
    }
    return NULL;
}

void hm_put(HashMap *hm, const char *key, int value) {
    unsigned int idx = hash(key, hm->capacity);
    Entry *exist = hm_find_entry(hm->buckets[idx], key);
    if (exist) {
        exist->value = value;
        return;
    }
    Entry *e = malloc(sizeof(Entry));
    e->key = strdup(key);
    e->value = value;
    e->next = hm->buckets[idx];
    hm->buckets[idx] = e;
    hm->size++;
    // 扩容逻辑略
}

int hm_get(HashMap *hm, const char *key, int *found) {
    unsigned int idx = hash(key, hm->capacity);
    Entry *e = hm_find_entry(hm->buckets[idx], key);
    if (e) {
        *found = 1;
        return e->value;
    }
    *found = 0;
    return -1;
}

int hm_remove(HashMap *hm, const char *key) {
    unsigned int idx = hash(key, hm->capacity);
    Entry *prev = NULL, *cur = hm->buckets[idx];
    while (cur) {
        if (strcmp(cur->key, key) == 0) {
            if (prev) prev->next = cur->next;
            else hm->buckets[idx] = cur->next;
            free(cur->key);
            free(cur);
            hm->size--;
            return 1;
        }
        prev = cur;
        cur = cur->next;
    }
    return 0;
}
```

### Java | 链地址法 (手动实现)

```java
class Entry {
    String key;
    int value;
    Entry next;
    Entry(String k, int v) { key = k; value = v; }
}

class MyHashMap {
    private Entry[] buckets;
    private int size;
    private static final int DEFAULT_CAP = 16;

    public MyHashMap() {
        buckets = new Entry[DEFAULT_CAP];
        size = 0;
    }

    private int hash(String key) {
        int h = 0;
        for (int i = 0; i < key.length(); i++)
            h = h * 31 + key.charAt(i);
        return (h & 0x7fffffff) % buckets.length;
    }

    // 插入/更新 O(1) 均摊
    public void put(String key, int value) {
        int idx = hash(key);
        // 查是否已存在
        for (Entry e = buckets[idx]; e != null; e = e.next) {
            if (e.key.equals(key)) {
                e.value = value;
                return;
            }
        }
        // 头插
        Entry e = new Entry(key, value);
        e.next = buckets[idx];
        buckets[idx] = e;
        size++;
    }

    // 查找 O(1) 均摊
    public Integer get(String key) {
        int idx = hash(key);
        for (Entry e = buckets[idx]; e != null; e = e.next)
            if (e.key.equals(key)) return e.value;
        return null;
    }

    // 删除 O(1) 均摊
    public boolean remove(String key) {
        int idx = hash(key);
        Entry prev = null, cur = buckets[idx];
        while (cur != null) {
            if (cur.key.equals(key)) {
                if (prev != null) prev.next = cur.next;
                else buckets[idx] = cur.next;
                size--;
                return true;
            }
            prev = cur;
            cur = cur.next;
        }
        return false;
    }

    public int size() { return size; }
    public boolean containsKey(String key) { return get(key) != null; }
}
```

> [!important] HashMap 源码要点 (JDK 8+)
> 1. 初始容量 16，负载因子 0.75
> 2. 链表长度 ≥ 8 且数组长度 ≥ 64 → 红黑树
> 3. 树节点 ≤ 6 → 转回链表
> 4. 扩容为 2 倍，rehash 优化高位判断
> 5. **非线程安全** → 用 `ConcurrentHashMap`

### 7.2 二叉搜索树 (BST)

**左 < 根 < 右。查找/插入/删除 平均 $O(\log n)$，最坏 $O(n)$。**

```c
// ===== C: BST 插入 & 查找 =====
TreeNode *bst_insert(TreeNode *root, int val) {
    if (!root) return tn_create(val);
    if (val < root->val)
        root->left = bst_insert(root->left, val);
    else if (val > root->val)
        root->right = bst_insert(root->right, val);
    return root;
}

TreeNode *bst_search(TreeNode *root, int val) {
    while (root) {
        if (val == root->val) return root;
        root = val < root->val ? root->left : root->right;
    }
    return NULL;
}

// BST 删除
TreeNode *bst_delete(TreeNode *root, int val) {
    if (!root) return NULL;
    if (val < root->val) {
        root->left = bst_delete(root->left, val);
    } else if (val > root->val) {
        root->right = bst_delete(root->right, val);
    } else {
        // 找到要删节点
        if (!root->left) {
            TreeNode *tmp = root->right;
            free(root);
            return tmp;
        }
        if (!root->right) {
            TreeNode *tmp = root->left;
            free(root);
            return tmp;
        }
        // 有两个子节点: 找后继(右子树最小值)
        TreeNode *succ = root->right;
        while (succ->left) succ = succ->left;
        root->val = succ->val;
        root->right = bst_delete(root->right, succ->val);
    }
    return root;
}
```

```java
class BST {
    TreeNode root;

    // 插入
    public void insert(int val) {
        root = insert(root, val);
    }
    private TreeNode insert(TreeNode node, int val) {
        if (node == null) return new TreeNode(val);
        if (val < node.val) node.left = insert(node.left, val);
        else if (val > node.val) node.right = insert(node.right, val);
        return node;
    }

    // 查找 (迭代)
    public TreeNode search(int val) {
        TreeNode cur = root;
        while (cur != null) {
            if (val == cur.val) return cur;
            cur = val < cur.val ? cur.left : cur.right;
        }
        return null;
    }

    // 删除
    public void delete(int val) {
        root = delete(root, val);
    }
    private TreeNode delete(TreeNode node, int val) {
        if (node == null) return null;
        if (val < node.val) {
            node.left = delete(node.left, val);
        } else if (val > node.val) {
            node.right = delete(node.right, val);
        } else {
            // 找到要删节点
            if (node.left == null) return node.right;
            if (node.right == null) return node.left;
            // 两个子节点: 找后继
            TreeNode succ = node.right;
            while (succ.left != null) succ = succ.left;
            node.val = succ.val;
            node.right = delete(node.right, succ.val);
        }
        return node;
    }

    // 验证 BST (中序遍历升序)
    public boolean isValidBST() {
        return isValidBST(root, null, null);
    }
    private boolean isValidBST(TreeNode node, Integer min, Integer max) {
        if (node == null) return true;
        if ((min != null && node.val <= min) || (max != null && node.val >= max))
            return false;
        return isValidBST(node.left, min, node.val)
            && isValidBST(node.right, node.val, max);
    }
}
```

### 7.3 堆 (Heap)

**完全二叉树，父节点 ≥ 子节点(最大堆) 或 ≤ (最小堆)。插入/删除 $O(\log n)$，建堆 $O(n)$。**

```c
// ===== C: 最大堆 =====
typedef struct {
    int *data;
    int size;
    int capacity;
} MaxHeap;

MaxHeap *heap_create(int cap) {
    MaxHeap *h = malloc(sizeof(MaxHeap));
    h->data = malloc(sizeof(int) * (cap + 1)); // 索引从1开始
    h->size = 0;
    h->capacity = cap;
    return h;
}

void heap_swim(MaxHeap *h, int k) {
    while (k > 1 && h->data[k] > h->data[k / 2]) {
        int tmp = h->data[k];
        h->data[k] = h->data[k / 2];
        h->data[k / 2] = tmp;
        k /= 2;
    }
}

void heap_sink(MaxHeap *h, int k) {
    while (2 * k <= h->size) {
        int j = 2 * k;  // 左孩子
        if (j < h->size && h->data[j] < h->data[j + 1]) j++; // 取大者
        if (h->data[k] >= h->data[j]) break;
        int tmp = h->data[k];
        h->data[k] = h->data[j];
        h->data[j] = tmp;
        k = j;
    }
}

void heap_insert(MaxHeap *h, int val) {
    h->data[++(h->size)] = val;
    heap_swim(h, h->size);
}

int heap_remove_max(MaxHeap *h) {
    int max = h->data[1];
    h->data[1] = h->data[h->size--];
    heap_sink(h, 1);
    return max;
}

// 建堆 O(n): 从最后一个非叶子节点往前下沉
void heap_build(MaxHeap *h, int arr[], int n) {
    h->size = n;
    for (int i = 0; i < n; i++) h->data[i + 1] = arr[i];
    for (int k = n / 2; k >= 1; k--) heap_sink(h, k);
}

// ===== 堆排序 O(n log n) =====
void heap_sort(int arr[], int n) {
    // 原地建最大堆
    for (int k = n / 2 - 1; k >= 0; k--) {
        // sink...
    }
    // 依次取最大
    for (int i = n - 1; i > 0; i--) {
        int tmp = arr[0];
        arr[0] = arr[i];
        arr[i] = tmp;
        // sink(0, i-1)...
    }
}
```

```java
class MaxHeap {
    private int[] data;
    private int size;
    private int capacity;

    public MaxHeap(int cap) {
        capacity = cap;
        data = new int[cap + 1]; // 索引从 1 开始
        size = 0;
    }

    // 上浮 O(log n)
    private void swim(int k) {
        while (k > 1 && data[k] > data[k / 2]) {
            swap(k, k / 2);
            k /= 2;
        }
    }

    // 下沉 O(log n)
    private void sink(int k) {
        while (2 * k <= size) {
            int j = 2 * k;
            if (j < size && data[j] < data[j + 1]) j++; // 取大孩子
            if (data[k] >= data[j]) break;
            swap(k, j);
            k = j;
        }
    }

    private void swap(int i, int j) {
        int tmp = data[i];
        data[i] = data[j];
        data[j] = tmp;
    }

    // 插入 O(log n)
    public void insert(int val) {
        data[++size] = val;
        swim(size);
    }

    // 删除堆顶 O(log n)
    public int removeMax() {
        int max = data[1];
        data[1] = data[size--];
        sink(1);
        return max;
    }

    public int peek() { return size > 0 ? data[1] : -1; }
    public int size() { return size; }

    // 建堆 O(n) — 从最后一个非叶子节点往前下沉
    public void buildHeap(int[] arr) {
        size = arr.length;
        for (int i = 0; i < arr.length; i++)
            data[i + 1] = arr[i];
        for (int k = size / 2; k >= 1; k--)
            sink(k);
    }
}
```

### 7.4 并查集 (Union-Find)

**近乎 $O(1)$ 判断连通性。路径压缩 + 按秩合并。**

```c
// ===== C: 并查集 =====
typedef struct {
    int *parent;
    int *rank; // 或 size
    int count; // 连通分量数
} UnionFind;

UnionFind *uf_create(int n) {
    UnionFind *uf = malloc(sizeof(UnionFind));
    uf->parent = malloc(sizeof(int) * n);
    uf->rank = calloc(n, sizeof(int));
    uf->count = n;
    for (int i = 0; i < n; i++) uf->parent[i] = i;
    return uf;
}

// 路径压缩
int uf_find(UnionFind *uf, int x) {
    if (uf->parent[x] != x)
        uf->parent[x] = uf_find(uf, uf->parent[x]); // 递归压缩
    return uf->parent[x];
}

// 按秩合并
void uf_union(UnionFind *uf, int x, int y) {
    int rx = uf_find(uf, x);
    int ry = uf_find(uf, y);
    if (rx == ry) return;
    if (uf->rank[rx] < uf->rank[ry]) {
        uf->parent[rx] = ry;
    } else if (uf->rank[rx] > uf->rank[ry]) {
        uf->parent[ry] = rx;
    } else {
        uf->parent[ry] = rx;
        uf->rank[rx]++;
    }
    uf->count--;
}

// 是否连通
int uf_connected(UnionFind *uf, int x, int y) {
    return uf_find(uf, x) == uf_find(uf, y);
}
```

```java
    private int[] parent;
    private int[] rank;
    public int count;

    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        count = n;
        for (int i = 0; i < n; i++) parent[i] = i;
    }

    public int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    public void union(int x, int y) {
        int rx = find(x), ry = find(y);
        if (rx == ry) return;
        if (rank[rx] < rank[ry]) {
            parent[rx] = ry;
        } else if (rank[rx] > rank[ry]) {
            parent[ry] = rx;
        } else {
            parent[ry] = rx;
            rank[rx]++;
        }
        count--;
    }

    public boolean connected(int x, int y) {
        return find(x) == find(y);
    }
}
```

### C | Dijkstra 最短路径

```c
#define INF 0x3f3f3f3f

void dijkstra(Graph *g, int start, int dist[]) {
    int visited[MAX_V] = {0};
    for (int i = 0; i < g->V; i++) dist[i] = INF;
    dist[start] = 0;

    for (int i = 0; i < g->V; i++) {
        int u = -1, min = INF;
        for (int j = 0; j < g->V; j++)
            if (!visited[j] && dist[j] < min)
                u = j, min = dist[j];
        if (u == -1) break;
        visited[u] = 1;
        for (Edge *e = g->adj[u].head; e; e = e->next)
            if (!visited[e->to] && dist[u] + e->weight < dist[e->to])
                dist[e->to] = dist[u] + e->weight;
    }
}
```

```java
                if (!visited[e.to]) stack[++top] = e.to;
        }
    }

    // 拓扑排序 (Kahn)
    public boolean topologicalSort() {
        int[] indegree = new int[V];
        for (int i = 0; i < V; i++)
            for (EdgeNode e = adj[i]; e != null; e = e.next)
                indegree[e.to]++;

        int[] queue = new int[V];
        int head = 0, tail = 0;
        for (int i = 0; i < V; i++)
            if (indegree[i] == 0) queue[tail++] = i;

        int count = 0;
        while (head < tail) {
            int v = queue[head++];
            System.out.print(v + " ");
            count++;
            for (EdgeNode e = adj[v]; e != null; e = e.next)
                if (--indegree[e.to] == 0)
                    queue[tail++] = e.to;
        }
        return count == V; // true = 无环
    }

    // Dijkstra (手动堆优化)
    public int[] dijkstra(int start) {
        int INF = Integer.MAX_VALUE;
        int[] dist = new int[V];
        boolean[] visited = new boolean[V];
        for (int i = 0; i < V; i++) dist[i] = INF;
        dist[start] = 0;

        for (int i = 0; i < V; i++) {
            int u = -1, min = INF;
            for (int j = 0; j < V; j++)
                if (!visited[j] && dist[j] < min) { u = j; min = dist[j]; }
            if (u == -1) break;
            visited[u] = true;
            for (EdgeNode e = adj[u]; e != null; e = e.next)
                if (!visited[e.to] && dist[u] != INF
                        && dist[u] + e.weight < dist[e.to])
                    dist[e.to] = dist[u] + e.weight;
        }
        return dist;
    }
}
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// ===== 快排 =====
int partition(int arr[], int lo, int hi) {
    int pivot = arr[lo];  // 或三数取中
    int i = lo, j = hi;
    while (i < j) {
        while (i < j && arr[j] >= pivot) j--;
        arr[i] = arr[j];
        while (i < j && arr[i] <= pivot) i++;
        arr[j] = arr[i];
    }
    arr[i] = pivot;
    return i;
}

void quick_sort(int arr[], int lo, int hi) {
    if (lo >= hi) return;
    int p = partition(arr, lo, hi);
    quick_sort(arr, lo, p - 1);
    quick_sort(arr, p + 1, hi);
}

// ===== 归并排序 =====
void merge(int arr[], int tmp[], int lo, int mid, int hi) {
    int i = lo, j = mid + 1, k = lo;
    while (i <= mid && j <= hi)
        tmp[k++] = arr[i] <= arr[j] ? arr[i++] : arr[j++];
    while (i <= mid)  tmp[k++] = arr[i++];
    while (j <= hi)   tmp[k++] = arr[j++];
    for (i = lo; i <= hi; i++) arr[i] = tmp[i];
}

void merge_sort(int arr[], int tmp[], int lo, int hi) {
    if (lo >= hi) return;
    int mid = lo + (hi - lo) / 2;
    merge_sort(arr, tmp, lo, mid);
    merge_sort(arr, tmp, mid + 1, hi);
    merge(arr, tmp, lo, mid, hi);
}

// ===== 冒泡 =====
void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int swapped = 0;
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
                swapped = 1;
            }
        }
        if (!swapped) break;  // 提前终止
    }
}

// ===== 插入排序 =====
void insertion_sort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

// ===== 选择排序 =====
void selection_sort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++)
            if (arr[j] < arr[min_idx]) min_idx = j;
        int tmp = arr[i];
        arr[i] = arr[min_idx];
        arr[min_idx] = tmp;
    }
}

// ===== 计数排序 (非比较) =====
void counting_sort(int arr[], int n) {
    int max = arr[0], min = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) max = arr[i];
        if (arr[i] < min) min = arr[i];
    }
    int range = max - min + 1;
    int *count = calloc(range, sizeof(int));
    for (int i = 0; i < n; i++) count[arr[i] - min]++;
    for (int i = 0, idx = 0; i < range; i++)
        while (count[i]-- > 0) arr[idx++] = i + min;
    free(count);
}
```

### Java 手动实现

```java
public class SortAlgos {
    // ===== 快速排序 =====
    public static void quickSort(int[] arr, int lo, int hi) {
        if (lo >= hi) return;
        int p = partition(arr, lo, hi);
        quickSort(arr, lo, p - 1);
        quickSort(arr, p + 1, hi);
    }
    private static int partition(int[] arr, int lo, int hi) {
        int pivot = arr[lo];
        int i = lo, j = hi;
        while (i < j) {
            while (i < j && arr[j] >= pivot) j--;
            arr[i] = arr[j];
            while (i < j && arr[i] <= pivot) i++;
            arr[j] = arr[i];
        }
        arr[i] = pivot;
        return i;
    }

    // ===== 归并排序 =====
    public static void mergeSort(int[] arr, int[] tmp, int lo, int hi) {
        if (lo >= hi) return;
        int mid = lo + (hi - lo) / 2;
        mergeSort(arr, tmp, lo, mid);
        mergeSort(arr, tmp, mid + 1, hi);
        merge(arr, tmp, lo, mid, hi);
    }
    private static void merge(int[] arr, int[] tmp, int lo, int mid, int hi) {
        int i = lo, j = mid + 1, k = lo;
        while (i <= mid && j <= hi)
            tmp[k++] = arr[i] <= arr[j] ? arr[i++] : arr[j++];
        while (i <= mid) tmp[k++] = arr[i++];
        while (j <= hi)  tmp[k++] = arr[j++];
        for (i = lo; i <= hi; i++) arr[i] = tmp[i];
    }

    // ===== 冒泡 =====
    public static void bubbleSort(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < arr.length - 1 - i; j++)
                if (arr[j] > arr[j + 1]) {
                    swap(arr, j, j + 1);
                    swapped = true;
                }
            if (!swapped) break;
        }
    }

    // ===== 插入 =====
    public static void insertionSort(int[] arr) {
        for (int i = 1; i < arr.length; i++) {
            int key = arr[i], j = i - 1;
            while (j >= 0 && arr[j] > key) arr[j + 1] = arr[j--];
            arr[j + 1] = key;
        }
    }

    // ===== 选择 =====
    public static void selectionSort(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < arr.length; j++)
                if (arr[j] < arr[minIdx]) minIdx = j;
            swap(arr, i, minIdx);
        }
    }

    // ===== 计数 =====
    public static void countingSort(int[] arr) {
        int max = arr[0], min = arr[0];
        for (int n : arr) {
            if (n > max) max = n;
            if (n < min) min = n;
        }
        int range = max - min + 1;
        int[] count = new int[range];
        for (int n : arr) count[n - min]++;
        int idx = 0;
        for (int i = 0; i < range; i++)
            while (count[i]-- > 0) arr[idx++] = i + min;
    }

    private static void swap(int[] arr, int i, int j) {
        int tmp = arr[i]; arr[i] = arr[j]; arr[j] = tmp;
    }
}
```

> [!tip] Dual-Pivot QuickSort vs TimSort
> - `Arrays.sort(int[])` → Dual-Pivot QuickSort，$O(n \log n)$，不稳定
> - `Arrays.sort(Object[])` → TimSort (归并+插入混合)，稳定，利用部分有序性
> - 对象排序选 TimSort 因为对象比较开销大，归并减少比较次数

```c
// ===== 二分查找 (升序数组) =====
int binary_search(int arr[], int n, int target) {
    int lo = 0, hi = n - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2; // 防溢出
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1;
}

// 二分查找左边界 (第一个 >= target)
int binary_left(int arr[], int n, int target) {
    int lo = 0, hi = n;
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (arr[mid] >= target) hi = mid;
        else lo = mid + 1;
    }
    return lo < n && arr[lo] == target ? lo : -1;
}

// 二分查找右边界 (最后一个 <= target)
int binary_right(int arr[], int n, int target) {
    int lo = -1, hi = n - 1;
    while (lo < hi) {
        int mid = lo + (hi - lo + 1) / 2;
        if (arr[mid] <= target) lo = mid;
        else hi = mid - 1;
    }
    return lo >= 0 && arr[lo] == target ? lo : -1;
}
```

```java
// 单调栈模板: 找下一个更大元素
int[] nextGreaterElement(int[] nums) {
    int n = nums.length;
    int[] res = new int[n];
    Arrays.fill(res, -1);
    Deque<Integer> stack = new ArrayDeque<>(); // 存下标
    for (int i = 0; i < n; i++) {
        while (!stack.isEmpty() && nums[stack.peek()] < nums[i])
            res[stack.pop()] = nums[i];
        stack.push(i);
    }
    return res;
}

// 滑动窗口模板: 最长无重复子串
int lengthOfLongestSubstring(String s) {
    int[] window = new int[128];
    int left = 0, maxLen = 0;
    for (int right = 0; right < s.length(); right++) {
        char c = s.charAt(right);
        window[c]++;
        while (window[c] > 1) {
            window[s.charAt(left)]--;
            left++;
        }
        maxLen = Math.max(maxLen, right - left + 1);
    }
    return maxLen;
}
```
